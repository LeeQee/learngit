git init

git add readme.txt
git commit -m "修改信息备注"

//修改readme.txt,查看修改状态和信息
git status

//提交修改
git add readme.txt
//查看状态
git status
//提交
git commit -m "修改备注信息"

//查看历史修改记录或者让提交版本
git log
git log --pretty==oneline
//版本回退
git reset --hard HEAD^
git reset --hard HEAD~100
git reset --hard 5bd44a
//查看回退版本记录
git reflog
//查看工作区和版本库的区别
git diff HEAD -- readme.txt
